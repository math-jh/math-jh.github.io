\documentclass[tikz,border=10pt]{standalone}
\usepackage{tikz,kotex}
\usepackage{tikz-3dplot}
\usetikzlibrary{calc}

\begin{document}

\tdplotsetmaincoords{70}{110}

\begin{tikzpicture}[tdplot_main_coords, scale=1.5]
  
  % Sphere (wireframe)
  \shade[ball color=white, opacity=0.3] (0,0,0) circle (2cm);
  
  % Great circle: x_2 = 0 (equator)
  \tdplotsetrotatedcoords{0}{0}{0}
  \draw[blue, thick, tdplot_rotated_coords] (0,0) circle (2);
  \node[blue] at (2.3, 0, 0) {$x_2 = 0$};
  
  % Great circle: x_0 = 0 (yz-plane)
  \tdplotsetrotatedcoords{90}{0}{0}
  \draw[green!60!black, thick, tdplot_rotated_coords] (0,0) circle (2);
  \node[green!60!black] at (0, 2.3, 0) {$x_0 = 0$};
  
  % Great circle: x_1 = 0 (xz-plane)
  \tdplotsetrotatedcoords{0}{90}{0}
  \draw[orange, thick, tdplot_rotated_coords] (0,0) circle (2);
  \node[orange] at (0, 0, 2.3) {$x_1 = 0$};
  
  % Latitude circle: X (z = 0.5 for example)
  \draw[red, very thick] (0,0,0.8) circle ({sqrt(4-0.64)});
  \node[red] at (1.8, 0, 1) {$X$};
  
  % Intersection points
  \fill[green!60!black] ({-sqrt(4-0.64)}, 0, 0.8) circle (2pt);
  \fill[green!60!black] ({sqrt(4-0.64)}, 0, 0.8) circle (2pt);
  
  \fill[orange] (0, {-sqrt(4-0.64)}, 0.8) circle (2pt);
  \fill[orange] (0, {sqrt(4-0.64)}, 0.8) circle (2pt);
  
  % Axes
  \draw[->, thick] (0,0,0) -- (2.5,0,0) node[anchor=north east]{$x_0$};
  \draw[->, thick] (0,0,0) -- (0,2.5,0) node[anchor=north west]{$x_1$};
  \draw[->, thick] (0,0,0) -- (0,0,2.5) node[anchor=south]{$x_2$};
  
  % Label
  \node at (0, 0, -3) {$S^2$ 위에서의 $X$와 무한대 직선들};
  
\end{tikzpicture}

\end{document}
