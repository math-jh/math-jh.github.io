% 왼쪽: 삼각형 강조

  \draw[ghost] (0,0) -- (\r,0) -- (\r,{\r*tan(\xa)}) -- cycle;
  % 강조 도형의 외골선 (가장 위)
  \draw[boundary] (0,0) -- (\r,0) -- ({\r*cos(\xa)},{\r*sin(\xa)}) -- cycle;

  \node at (0.72,0.22) {$x$};
  \node[below] at (0.7,-0.55) {$\frac{1}{2}\sin x$};
\end{scope}
