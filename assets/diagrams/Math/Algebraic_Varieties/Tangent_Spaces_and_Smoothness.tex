    >=Stealth,
    axis/.style={->,thick,black!50},
    curve/.style={thick,accent1,smooth}
]
    \draw[axis] (-2.2,0) -- (2.5,0) node[right] {$\x$};
    \draw[axis] (0,-2.2) -- (0,2.5) node[above] {$\y$};
    \draw[curve,domain=-1:1.5,samples=100,variable=\x]
        plot ({\x},{\x*sqrt(\x+1)});
    \draw[curve,domain=-1:1.5,samples=100,variable=\x]
        plot ({\x},{-\x*sqrt(\x+1)});

    \draw[dashed,accent3,thick] (-1.5,-1.5) -- (1.5,1.5)
        node[pos=0.85,below right] {$\y=\x$};
    \draw[dashed,accent3,thick] (-1.5,1.5) -- (1.5,-1.5)
        node[pos=0.85,above right] {$\y=-\x$};
    \node[accent1] at (1.8,2.6) {$X: \y^2 = \x^2(\x+1)$};
\end{tikzpicture}
